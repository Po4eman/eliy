script.
  var slides;
  document.addEventListener("DOMContentLoaded", ()=>{
  slides = document.querySelector('#mySlider>.slides').children;
  for(var i = 0, slide; slide = slides[i]; i++){
  slide.setAttribute("slide-cnt", i);
  }
  slides[0].style.opacity = 1;
  slideShow();
  });
  var slidecnt = 0;
  function slideShow(){
  slides[slidecnt].style.opacity = 0;
  slides[slidecnt].style.maxWidth = "0%";
  if(slidecnt >= slides.length-1){
  slidecnt = 0;
  }else{
  slidecnt++;
  }
  slides[slidecnt].style.opacity = 1;
  slides[slidecnt].style.maxWidth = "100%";
  console.log(slidecnt, slides.length);
  setTimeout(slideShow,5000);
  }
